115 template <typename T, std::size_t M, std::size_t N>
116 constexpr std::tuple<matrix<T, M, N>, std::size_t, T>
117 gauss_jordan_impl(matrix<T, M, N> m, T tolerance) {
118  // Define function for determining if an element is negligible
119  auto negligible = [&tolerance](const T &v) { return abs(v) < tolerance; };
120 
121  T det = 1;
122  std::size_t rank = 0;
123  std::size_t i = 0, j = 0;
124  while (i < M && j < N) {
125  // Choose largest magnitude as pivot to avoid adding different magnitudes
126  for (std::size_t ip = i + 1; ip < M; ++ip) {
127  if (abs(m[ip][j]) > abs(m[i][j])) {
128  for (std::size_t jp = 0; jp < N; ++jp) {
129  auto tmp = m[ip][jp];
130  m[ip][jp] = m[i][jp];
131  m[i][jp] = tmp;
132  }
133  det *= -1;
134  break;
135  }
136  }
137 
138  // If m_ij is still 0, continue to the next column
139  if (!negligible(m[i][j])) {
140  // Scale m_ij to 1
141  auto s = m[i][j];
142  for (std::size_t jp = 0; jp < N; ++jp)
143  m[i][jp] /= s;
144  det /= s;
145 
146  // Eliminate other values in the column
147  for (std::size_t ip = 0; ip < M; ++ip) {
148  if (ip == i)
149  continue;
150  if (!negligible(m[ip][j])) {
151  auto s = m[ip][j];
152  [&]() { // wrap this in a lambda to get around a gcc bug
153  for (std::size_t jp = 0; jp < N; ++jp)
154  m[ip][jp] -= s * m[i][jp];
155  }();
156  }
157  }
158 
159  // Increment rank
160  ++rank;
161 
162  // Select next row
163  ++i;
164  }
165  ++j;
166  }
167  det = (rank == M) ? det : 0;
168  return {m, rank, det};
169 }
