92 constexpr matrix<T, M, N> gauss_elim(matrix<T, M, M> a,
93  matrix<T, M, N> adj) {
94  for (std::size_t k = 0; k < M; ++k) {
95  // first normalize the row
96  auto scale_A = a[k][k];
97  if (scale_A == 0)
98  throw "Matrix on left hand side is not invertible";
99 
100  for (std::size_t j = 0; j < N; ++j) {
101  adj[k][j] = adj[k][j] / scale_A;
102  a[k][j] = a[k][j] / scale_A;
103  }
104  auto pivot_A = a.row(k);
105  auto pivot_adj = adj.row(k);
106 
107  for (std::size_t i = 0; i < M; ++i) {
108  if (i != k) {
109  // pivot should be normalized to 1 already
110  auto row_to_subtract_A = pivot_A * a[i][k];
111  auto row_to_subtract_adj = pivot_adj * a[i][k];
112  for (std::size_t j = 0; j < M; ++j) {
113  a[i][j] = a[i][j] - row_to_subtract_A[j];
114  }
115  for (std::size_t j = 0; j < N; ++j) {
116  adj[i][j] = adj[i][j] - row_to_subtract_adj[j];
117  }
